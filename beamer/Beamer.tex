\subsection{Entwickeln eines Prototypen für die Verwaltung von Normen und Maßnahmen}
\begin{frame}{Entwickeln eines Prototypen für die \\ Verwaltung von Normen und Maßnahmen}
	\begin{tikzpicture}[remember picture,overlay]
		\node[anchor=north east, xshift=-0.2cm, yshift=-0.2cm] at (current page.north east) {
			\includegraphics[width=2cm]{arca.png}
		};
	\end{tikzpicture}
	\begin{onlyenv}<1>
		\begin{itemize}
			\item Früher: Next.js, Ohne REST-API, Direkter Datenbankzugriff im Backend
			\item Next.js verursachte Probleme:
			      \begin{itemize}
				      \item langsamer Build
				      \item zu aggressives Caching
				      \item unpassende Defaults
				      \item schwer vorhersehbarer Controlflow
			      \end{itemize}
		\end{itemize}
	\end{onlyenv}

	\begin{onlyenv}<2>
		\begin{itemize}
			\item Neu: klare Trennung Frontend / Backend
			\item \textbf{Frontend}: React + TanStack Router + TanStack Query
			\item Vorteile:
			      \begin{itemize}
				      \item Schnellerer Build (2+min $\rightarrow$ 10s)
				      \item bessere Caching-Strategien, mehr Kontrolle über Routing und Daten-Fetching
				      \item Wiederverwenden von vielen Componenten des alten Prototyps
				      \item Automatisierte Generierung von API-Clients mit orval.js
			      \end{itemize}
			\item Nachteile:
			      \begin{itemize}
				      \item Managen von API Specs und Code-Generation
				      \item Anpassung von Komponenten an neue Daten-Fetching-Strategien
				            \customnote<2>{Ein API endpoint anstatt spezifischen DB Queries pro component}
			      \end{itemize}
			      \vspace{1.5cm}
		\end{itemize}
	\end{onlyenv}

	\begin{onlyenv}<3>
		\begin{itemize}
			\item \textbf{Backend}: JVM, Ktor als Framework, JOOQ als ORM
			\item Sprache: Kotlin statt Java
			      \begin{itemize}
				      \item Kürzere, ausdrucksstarke Syntax, weniger Boilerplate
				      \item Null-Safety und Coroutines
				      \item Gute Interoperabilität mit Java-Ökosystem und Libraries
				      \item Flexible Programmierparadigmen % (funktional, objektorientiert, prozedural)
			      \end{itemize}
		\end{itemize}
	\end{onlyenv}

	\begin{onlyenv}<4>
		\begin{itemize}
			\item \textbf{Datenzugriff}: verschiedene ORMs / Query-Builder ausprobiert
			      \begin{itemize}
				      \item Kotlin Exposed
				            \customnote<4>{Kein R2DBC Support}
				      \item Komapper
				            \customnote<4>{Beide, schema in code, generiert migrations}
				      \item JOOQ, Migrations mit Flyway
				            \customnote<4>{generiert Kotlin interfaces basierend auf DB schema}
			      \end{itemize}

			\item \textbf{Observability}: OpenTelemetry-Integration \\(Logs, Traces, Metrics)\\
			      End-to-end Tracing (Frontend $\rightarrow$ Backend $\rightarrow$ DB)
			\item \textbf{OpenAPI}: smiley4 ktor-openapi, Natives OpenAPI-Support in Ktor derzeit experimentell
			      \customnote<4>{Eigenes Fazit, macht spaß, gut dass kleine firma}
		\end{itemize}
	\end{onlyenv}
